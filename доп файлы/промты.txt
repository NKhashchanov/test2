Стэк fastapi, js, postgresql.
будем создавать сайт с тренажером по PostgreSQL.
Задача на сегодня - создать главную страницу. 
Стиль минималистичный, строгий в серых тонах с небольшим добавлением оранжевого.
На странице должны быть следующие элементы:
- кнопка логина (можно сразу отрисовать страницу с формой логина)
- кнопка регистрации (можно сразу отрисовать страницу с формой регистрации)
- После логина должна быть кнопка личного кабинета (можно сразу сделать страницу заглушку)
Над таблицей со списком заданий должен быть выбор уровня сложности задания (хранится ва БД), темы задания (хранится ва БД).
В зависимости от выбранной сложности и темы происходит выборка из БД и заполнятся таблица с заданиями.
Список заданий должен быть в табличной форме, сами задания хранятся в БД.
- сложность, id задания, название задания, флаг выполнено или нет
Должна быть сортировка по сложности, названию, флагу выполнения.
Должна быть кнопка отображения кол-ва задач на странице (25, 50, 100), соответственно используем поганацию при получении задач из БД.






create table task_difficulty (
	task_difficulty_id serial primary key,
	task_difficulty_name text not null unique
)

create table task_theme (
	task_theme_id serial primary key,
	task_theme_name text not null unique
)

create table exercises (
	exercises_id serial primary key,
	exercises_name text not null unique,
	description text,
	task_difficulty_id int not null references task_difficulty(task_difficulty_id),
	task_theme_id int not null references task_theme(task_theme_id),
	query_template text not null,
	created_at timestamp not null default current_timestamp
)

create table user_exercises (
	user_exercises_id serial primary key,
	user_id int not null references users(user_id),
	exercise_id int not null references exercises(exercises_id),
	completed bool not null,
	completed_at timestamp not null default current_timestamp,
	user_query text not null
)

create table users (
	user_id serial primary key,
	email text not null,
	username varchar not null,
	hashed_password varchar not null,
	created_at timestamp not null default current_timestamp
)
